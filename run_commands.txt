# =============================================================================
#  Available CLI flags reference
# =============================================================================
#
#  run_all_evaluations.py:
#    --models          Model HF IDs (default: all 3)
#    --approaches      default | DynamicCheatsheet_Cumulative
#    --datasets        Dataset names (default: all 8)
#    --backend         hf | vllm
#    --quantization    none | 4bit | 8bit
#    --max_tokens      Max generation tokens (default: 2048)
#    --temperature     Sampling temperature (default: 0.0)
#    --max_num_rounds  Cheatsheet rounds (default: 1)
#    --max_samples     Samples per dataset (-1 = all, default: 15)
#    --no_code_execution   Disable code execution
#    --save_dir        Output directory (default: results_oss)
#    --resume          Skip completed runs
#    --no_parallel     Force single-GPU sequential mode
#    --cheatsheet_verbose  Save cheatsheet audit snapshots to <save_dir>/cheatsheet_audit/
#    --dc_batch_size   DC-Cumulative: questions per curator update (default: 32).
#                      Use 1 for per-question curation (good for auditing).
#
#  run_multi_agent_eval.py:
#    --approaches      MultiGenerator | MultiGenerator_Cumulative
#    --datasets        Dataset names (default: all)
#    --backend         hf | vllm
#    --quantization    none | 4bit | 8bit
#    --max_tokens      Max generation tokens (default: 2048)
#    --temperature     Sampling temperature (default: 0.0)
#    --max_num_rounds  Curator rounds (default: 1)
#    --max_samples     Samples per dataset (-1 = all, default: 15)
#    --no_code_execution   Disable code execution
#    --save_dir        Output directory (default: results_multi_agent)
#    --resume          Skip completed runs
#    --cheatsheet_verbose  Save cheatsheet audit snapshots (MultiGenerator_Cumulative only)
#    --dc_batch_size   MultiGenerator_Cumulative: questions per curator update (default: 32).
#                      Use 1 for per-question curation (good for auditing).
#
#  Available datasets:
#    AIME_2024, AIME_2025, AIME_2020_2024, GPQA_Diamond,
